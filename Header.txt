Requires the header CSS in Design > Custom CSS.

<script>
(function(){

  const FH_WORKER_BASE = "https://fletschhorn-guesty-api.bookings-e2d.workers.dev";
  const FH_INQUIRY_ENDPOINT = FH_WORKER_BASE.replace(/\/$/, "") + "/api/inquiry";

  window.FH_WORKER_BASE = FH_WORKER_BASE;
  window.FH_WORKER_ENDPOINT = FH_WORKER_BASE;
  window.FH_INQUIRY_ENDPOINT = FH_INQUIRY_ENDPOINT;
  window.CLOUDFLARE_INQUIRY_ENDPOINT = FH_INQUIRY_ENDPOINT;

  const HIDE_SQUARESPACE_NATIVE_HEADER_FOOTER = true;

  const FH_PAGES = [
    {
      key:'home',
      label:'Home',
      short:'Home',
      href:'/home',
      note:'Start with the house, setting, and inquiry.',
      match:['/','/Home','/home']
    },
    {
      key:'property',
      label:'The Property',
      short:'Property',
      href:'/the-property',
      note:'The property, spaces, spa, amenities, dining and kitchen.',
      match:['/the-property','/property','/spa-amenities','/dining-kitchen']
    },
    {
      key:'experiences',
      label:'Experiences',
      short:'Experiences',
      href:'/experiences',
      note:'What to do in Saas-Fee and around the property.',
      match:['/experiences','/what-to-do','/area']
    },
    {
      key:'events',
      label:'Wedding & Events',
      short:'Events',
      href:'/your-wedding-or-event',
      note:'Weddings, private events, retreats, and celebrations.',
      match:['/your-wedding-or-event','/wedding-and-events','/weddings-events','/weddings','/events']
    },
     {
      key:'location',
      label:'Location',
      short:'Location',
      href:'/location',
      note:'Where Fletschhorn is located.',
      match:['/Location','/where','/area']
    },
    {
      key:'booking',
      label:'Booking',
      short:'Booking',
      href:'/booking',
      note:'Send an inquiry and plan your stay or event.',
      match:['/booking','/contact-booking','/contact']
    },
    {
      key:'about',
      label:'About Us',
      short:'About',
      href:'/about-us',
      note:'Story, history, and the new private-house direction.',
      match:['/about-us','/about']
    }
  ];

  function normalizePath(path){
    let clean = (path || '/').split('?')[0].split('#')[0].replace(/\/+/g,'/');
    if(clean.length > 1) clean = clean.replace(/\/$/,'');
    return clean || '/';
  }

  function currentPageKey(){
    const bodyPage = document.querySelector('[data-fh-page], [data-page]');
    const explicit = bodyPage && (bodyPage.getAttribute('data-fh-page') || bodyPage.getAttribute('data-page'));

    if(explicit){
      const key = explicit.toLowerCase();
      if(key.includes('booking')) return 'booking';
      if(key.includes('event') || key.includes('wedding')) return 'events';
      if(key.includes('experience')) return 'experiences';
      if(key.includes('property')) return 'property';
      if(key.includes('about')) return 'about';
      if(key.includes('home')) return 'home';
    }

    const ids = [
      'fh-home',
      'fh-property',
      'fh-experiences',
      'fh-events',
      'fh-booking',
      'fh-about'
    ];

    for(const id of ids){
      if(document.getElementById(id)){
        return id.replace('fh-','');
      }
    }

    const path = normalizePath(window.location.pathname).toLowerCase();
    const found = FH_PAGES.find(page =>
      page.match.some(match => normalizePath(match).toLowerCase() === path)
    );

    return found ? found.key : 'home';
  }

  function makeHeader(activeKey){
    const navLinks = FH_PAGES.map(page => `
      <a href="${page.href}" class="${page.key === activeKey ? 'is-active' : ''}" data-fh-page-link="${page.key}">
        ${page.short}
      </a>
    `).join('');

    return `
      <header class="fh-global-header" id="fhGlobalHeader">
        <a class="fh-global-brand" href="/Home" aria-label="Fletschhorn home">
          <span class="fh-global-mark">
            <img src="/s/Saas-Fee-Logo.jpeg" alt="Fletschhorn" onerror="this.style.display='none'; this.parentElement.textContent='fh';" />
          </span>
          <span>
            <span class="fh-brand-name">Fletschhorn</span>
            <span class="fh-brand-note">Private alpine house · Saas-Fee</span>
          </span>
        </a>

        <nav class="fh-global-nav" aria-label="Primary navigation">
          ${navLinks}
          <a class="fh-global-cta" href="/booking">Book Now</a>
        </nav>

        <button class="fh-mobile-toggle" type="button" id="fhMobileToggle" aria-expanded="false" aria-controls="fhMobileMenu" aria-label="Open menu">
          <span></span>
        </button>
      </header>
    `;
  }

  function makeMobileMenu(activeKey){
    const links = FH_PAGES.map(page => `
      <a href="${page.href}" class="${page.key === activeKey ? 'is-active' : ''}">
        <span>
          <strong>${page.label}</strong>
          <small>${page.note}</small>
        </span>
        <em>→</em>
      </a>
    `).join('');

    return `
      <div class="fh-mobile-menu" id="fhMobileMenu" aria-hidden="true">
        <div class="fh-mobile-panel">
          <div class="fh-mobile-links">
            ${links}
          </div>
        </div>
      </div>
    `;
  }

  function init(){
    if(document.getElementById('fhGlobalHeader')) return;

    const activeKey = currentPageKey();

    document.body.classList.add('fh-global-shell');

    if(HIDE_SQUARESPACE_NATIVE_HEADER_FOOTER){
      document.body.classList.add('fh-hide-sqs-shell');
    }

    document.body.insertAdjacentHTML('afterbegin', makeMobileMenu(activeKey));
    document.body.insertAdjacentHTML('afterbegin', makeHeader(activeKey));

    const header = document.getElementById('fhGlobalHeader');
    const toggle = document.getElementById('fhMobileToggle');
    const menu = document.getElementById('fhMobileMenu');

    function onScroll(){
      if(!header) return;
      header.classList.toggle('is-scrolled', window.scrollY > 20);
    }

    function closeMenu(){
      if(!toggle || !menu) return;
      toggle.setAttribute('aria-expanded','false');
      menu.setAttribute('aria-hidden','true');
      menu.classList.remove('is-open');
      document.documentElement.classList.remove('fh-nav-open');
    }

    function openMenu(){
      if(!toggle || !menu) return;
      toggle.setAttribute('aria-expanded','true');
      menu.setAttribute('aria-hidden','false');
      menu.classList.add('is-open');
      document.documentElement.classList.add('fh-nav-open');
    }

    if(toggle && menu){
      toggle.addEventListener('click', () => {
        const isOpen = toggle.getAttribute('aria-expanded') === 'true';
        isOpen ? closeMenu() : openMenu();
      });

      menu.querySelectorAll('a').forEach(a => a.addEventListener('click', closeMenu));

      document.addEventListener('keydown', event => {
        if(event.key === 'Escape') closeMenu();
      });
    }

    window.addEventListener('scroll', onScroll, {passive:true});
    onScroll();
  }

  if(document.readyState === 'loading'){
    document.addEventListener('DOMContentLoaded', init);
  }else{
    init();
  }

})();
</script>
